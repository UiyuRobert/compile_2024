int g(int arr[]) {
    return arr[0]+arr[arr[1]+-arr[0]];
}

int main() {
    char c = 'a';
    c = getchar();
    int i;
    int array[3] = {1, 2, 3};
    i = g(array);

    for (c='a'; c < (127 % 128) || i != 0; c = c+1) {
        c=c+1;
        if (c=='x'){
            break;
        }
    }


    return 0;
}
